**Static good point 
class Outer {
    static class Inner {
        void display() {
            System.out.println("Static inner class");
        }
    }
}

public class Main {
    public static void main(String[] args) {
        Outer.Inner obj = new Outer.Inner(); // No need of Outer object
        obj.display();
    }
}

class Outer {
    class Inner {
        void display() {
            System.out.println("Non-static inner class");
        }
    }
}

public class Main {
    public static void main(String[] args) {
        Outer outerObj = new Outer();      // Step 1: Create Outer object
        Outer.Inner innerObj = outerObj.new Inner(); // Step 2: Use it to create Inner object
        innerObj.display();                // Output: Non-static inner class
    }
}
